function parseCode(text) { const manualSub = document.getElementById('manual-sub').value.trim(); const clean = text.toUpperCase().replace(/[^A-Z0-9_]/g, ' ').trim(); let resText = ""; let pap = "", varnt = "", seas = "", yr = "", sub = manualSub; let matchedStrict = false; // ---------------------------------------------------- // 1. STRICT MATCHING (Original fast patterns) // ---------------------------------------------------- const mUnder = clean.match(/(\d{4})?_?([MSW])(\d{2})_(QP|MS)_(\d)(\d)?/); const mShort = clean.match(/([MSW])\s*(\d{2})\s*[_ ]?\s*(QP|MS)\s*[_ ]?\s*(\d)(\d)?/); const mSlash = clean.match(/(\d{4})?\s*([1-8])\s*([1-6])?\s*(M\s*J|O\s*N|F\s*M|S|W|M)\s*(?:20)?(\d{2})/); if (mUnder || mShort) { const match = mUnder || mShort; if (!sub && mUnder && mUnder[1]) sub = mUnder[1]; seas = match[mUnder ? 2 : 1].toLowerCase(); yr = match[mUnder ? 3 : 2]; pap = match[mUnder ? 5 : 4]; varnt = match[mUnder ? 6 : 5] || ""; matchedStrict = true; } else if (mSlash) { if (!sub && mSlash[1]) sub = mSlash[1]; pap = mSlash[2]; varnt = mSlash[3] || ""; let seasStr = mSlash[4].replace(/\s/g, ''); yr = mSlash[5]; if (seasStr === 'MJ' || seasStr === 'S') seas = 's'; else if (seasStr === 'ON' || seasStr === 'W') seas = 'w'; else if (seasStr === 'FM' || seasStr === 'M') seas = 'm'; matchedStrict = true; } // ---------------------------------------------------- // 2. SMART FALLBACK (The puzzle-piece logic) // ---------------------------------------------------- if (!matchedStrict) { const smartClean = clean.replace(/_/g, ' ').replace(/\s+/g, ' '); // Normalize spaces // A. Find Subject (Any 4 digit number that isn't a year like 2014) if (!sub) { let subMatch = smartClean.match(/\b([0789]\d{3})\b/); if (subMatch) sub = subMatch[1]; } // B. Find Session if (/\b(M\s*J|S)\b/.test(smartClean)) seas = 's'; else if (/\b(O\s*N|W)\b/.test(smartClean)) seas = 'w'; else if (/\b(F\s*M|M)\b/.test(smartClean)) seas = 'm'; // C. Find Year (Starts with 20xx OR comes immediately after session like 'MJ 08') let fullYear = smartClean.match(/\b(?:20)(\d{2})\b/); if (fullYear) { yr = fullYear[1]; } else { let yrAfterSeas = smartClean.match(/\b(?:M\s*J|O\s*N|F\s*M|S|W|M)\s*(\d{2})\b/); if (yrAfterSeas) yr = yrAfterSeas[1]; } // D. Find Paper/Variant (Looks near 'QP/MS' OR before the session OR just finds remaining numbers) let qpMatch = smartClean.match(/\b(?:QP|MS|PAPER)\s*(\d{1,2})\b/); let papBeforeSeas = smartClean.match(/\b(\d{1,2})\s*(?:M\s*J|O\s*N|F\s*M|S|W|M)\b/); if (qpMatch) { pap = qpMatch[1][0]; varnt = qpMatch[1].length > 1 ? qpMatch[1][1] : ""; } else if (papBeforeSeas) { pap = papBeforeSeas[1][0]; varnt = papBeforeSeas[1].length > 1 ? papBeforeSeas[1][1] : ""; } else { // Ultimate fallback: Find any remaining 1-2 digit number that IS NOT the year let looseNums = smartClean.match(/\b\d{1,2}\b/g) || []; looseNums = looseNums.filter(n => n !== yr); // Remove the year so it doesn't get confused if (looseNums.length > 0) { pap = looseNums[0][0]; varnt = looseNums[0].length > 1 ? looseNums[0][1] : ""; } } } // ---------------------------------------------------- // 3. FINAL COMPILATION & PRE-2010 RULE // ---------------------------------------------------- sub = sub || "9709"; // Default if absolutely nothing was found if (seas && pap && yr) { let yrInt = parseInt(yr); let sName = (seas === 'm') ? "March" : (seas === 's' ? "Summer" : "Winter"); let fullYearStr = (yrInt > 50 ? "19" : "20") + yr; // The Time-Travel Rule: let finalVarnt = ""; if (yrInt < 10) { finalVarnt = ""; // Pre-2010 = Wipe the variant } else { finalVarnt = varnt || "1"; // Post-2010 = Keep it, or default to 1 } resText = `Paper ${pap}${finalVarnt ? " Variant " + finalVarnt : ""} ${sName} ${fullYearStr}`; subjectEl.textContent = `Subject: ${sub}`; resultEl.textContent = resText; const fileSuffix = pap + finalVarnt; const msUrlPart = `${sub}_${seas}${yr}_ms_${fileSuffix}`; const qpUrlPart = `${sub}_${seas}${yr}_qp_${fileSuffix}`; msLink.href = `https://pastpapers.papacambridge.com/directories/CAIE/CAIE-pastpapers/upload/${msUrlPart}.pdf`; qpLink.href = `https://pastpapers.papacambridge.com/directories/CAIE/CAIE-pastpapers/upload/${qpUrlPart}.pdf`; pdfContainer.style.display = 'flex'; if(autoCopy) navigator.clipboard.writeText(resText); } else { resultEl.textContent = "No valid code found"; pdfContainer.style.display = 'none'; } }
